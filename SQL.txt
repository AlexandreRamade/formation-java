
IFNULL(   ,   ) remplace les valeurs nules par la valeur indiquée en deuxième paramètre
(ISNULL(   ,   ) pour SqlServer - NVL(   ,   ) pour Oracle)

SELECT NOM, PRENOM, DATE_NAISSANCE, IFNULL(COMMISSION, 0) AS commission FROM employes;



TIMESTAMPSDIFF(   ,   ,   ) calcule la diférence entre deux dates et retourne la valeur dans l'unité souhaitée

SELECT NOM, PRENOM, TIMESTAMPDIFF(YEAR, DATE_NAISSANCE, CURDATE()) AS Age FROM employes;
